```typescript
import * as fs from 'node:fs';
import * as path from 'node:path';
import Ajv from 'ajv';
import type { MinionRegistry, SubprojectRegistryEntry } from '@minion/env';

const ajv = new Ajv();

export function loadRegistry(registryPath: string): MinionRegistry {
  const raw = JSON.parse(fs.readFileSync(registryPath, 'utf8'));
  const schemaPath = path.join(path.dirname(registryPath), 'packages', 'cli', 'minion.schema.json');
  if (fs.existsSync(schemaPath)) {
    const schema = JSON.parse(fs.readFileSync(schemaPath, 'utf8'));
    const validate = ajv.compile(schema);
    if (!validate(raw)) {
      const msg = (validate.errors ?? []).map(e => `${e.instancePath} ${e.message}`).join('; ');
      throw new Error(`minion.json invalid: ${msg}`);
    }
  } else {
    // Minimal structural check if schema not findable
    if (!raw || typeof raw !== 'object' || !raw.subprojects) {
      throw new Error('minion.json missing `subprojects`');
    }
    for (const [id, entry] of Object.entries(raw.subprojects as Record<string, unknown>)) {
      const e = entry as Partial<SubprojectRegistryEntry>;
      for (const field of ['path', 'packageManager', 'branch', 'infisicalProject', 'remote', 'commands'] as const) {
        if (e[field] === undefined) throw new Error(`Subproject '${id}' missing required field '${field}'`);
      }
    }
  }
  return raw;
}

export function getSubproject(reg: MinionRegistry, id: string): SubprojectRegistryEntry {
  const entry = reg.subprojects[id];
  if (!entry) {
    throw new Error(`Subproject '${id}' not found in minion.json. Run 'minion list' to see available ids.`);
  }
  return entry;
}

export function findMetaRoot(start: string = process.cwd()): string {
  let dir = path.resolve(start);
  for (let i = 0; i < 10; i++) {
    if (fs.existsSync(path.join(dir, 'minion.json'))) return dir;
    const parent = path.dirname(dir);
    if (parent === dir) break;
    dir = parent;
  }
  throw new Error(`minion.json not found upward from ${start}`);
}
```

```typescript
export function serialiseDotenv(env: Record<string, string>): string {
  const lines: string[] = [];
  for (const [k, v] of Object.entries(env)) {
    lines.push(`${k}=${encode(v)}`);
  }
  return lines.join('\n') + '\n';
}

function encode(v: string): string {
  if (v === '') return '';
  const needsQuotes = /[\s"'`$\\\n\r]/.test(v);
  if (!needsQuotes) return v;
  const escaped = v.replace(/\\/g, '\\\\').replace(/"/g, '\\"').replace(/\n/g, '\\n').replace(/\r/g, '\\r');
  return `"${escaped}"`;
}
```

`packages/cli/src/commands/fanout.ts`:
```typescript
import * as path from 'node:path';
import { execa } from 'execa';
import { resolveEnv } from '@minion/env';
import { findMetaRoot, loadRegistry } from '../registry';

export async function fanoutCommand(subcommand: 'dev' | 'build' | 'test' | 'check'): Promise<number> {
  const metaRoot = findMetaRoot();
  const reg = loadRegistry(path.join(metaRoot, 'minion.json'));
  const names = Object.keys(reg.subprojects).filter(id => reg.subprojects[id]!.commands[subcommand]);
  if (names.length === 0) { console.error(`No subprojects declare '${subcommand}'`); return 2; }
  // Build concurrently args: -n name1,name2 "cmd1" "cmd2"
  const cmds: string[] = [];
  for (const id of names) {
    const entry = reg.subprojects[id]!;
    const { env } = await resolveEnv({ subprojectId: id, cwd: metaRoot });
    const envPrefix = Object.entries(env).map(([k, v]) => `${k}=${shellEscape(v)}`).join(' ');
    cmds.push(`cd ${path.join(metaRoot, entry.path)} && ${envPrefix} ${entry.commands[subcommand]}`);
  }
  const args = ['-n', names.join(','), ...cmds];
  const result = await execa('concurrently', args, { stdio: 'inherit', preferLocal: true });
  return result.exitCode ?? 1;
}

function shellEscape(v: string): string {
  return "'" + v.replace(/'/g, "'\\''") + "'";
}
```

```typescript
import * as path from 'node:path';
import * as fs from 'node:fs';
import { execa } from 'execa';
import { findMetaRoot, loadRegistry } from '../registry';
import { printTable, printJson } from '../lib/output';

export async function statusCommand(json: boolean): Promise<number> {
  const metaRoot = findMetaRoot();
  const reg = loadRegistry(path.join(metaRoot, 'minion.json'));
  const rows: Array<Record<string, string>> = [];
  for (const [id, entry] of Object.entries(reg.subprojects)) {
    const subPath = path.join(metaRoot, entry.path);
    if (!fs.existsSync(path.join(subPath, '.git'))) {
      rows.push({ id, branch: '(no .git)', dirty: '-', ahead: '-', behind: '-' });
      continue;
    }
    const branch = (await safeExec('git', ['-C', subPath, 'rev-parse', '--abbrev-ref', 'HEAD'])).trim();
    const porcelain = await safeExec('git', ['-C', subPath, 'status', '--porcelain']);
    const dirty = porcelain.trim() ? String(porcelain.trim().split('\n').length) : '0';
    let ahead = '-', behind = '-';
    const counts = await safeExec('git', ['-C', subPath, 'rev-list', '--left-right', '--count', `${branch}...origin/${branch}`]);
    const m = counts.trim().match(/^(\d+)\s+(\d+)$/);
    if (m) { ahead = m[1]!; behind = m[2]!; }
    rows.push({ id, branch, dirty, ahead, behind });
  }
  if (json) printJson(rows); else printTable(rows);
  return 0;
}

async function safeExec(exe: string, args: string[]): Promise<string> {
  try {
    const r = await execa(exe, args, { reject: false });
    return r.stdout ?? '';
  } catch { return ''; }
}
```

```typescript
import { execa } from 'execa';
import { resolveEnv } from '@minion/env';
import { findMetaRoot, loadRegistry } from '../registry';
import { printTable, printJson } from '../lib/output';

export async function doctorCommand(json: boolean): Promise<number> {
  const metaRoot = findMetaRoot();
  const reg = loadRegistry(path.join(metaRoot, 'minion.json'));
  const rows: Array<Record<string, string>> = [];
  // Infisical auth probe
  const infisicalAuth = (process.env.INFISICAL_UNIVERSAL_AUTH_CLIENT_ID && process.env.INFISICAL_UNIVERSAL_AUTH_CLIENT_SECRET) ? 'ok' : 'missing';
  const infisicalBin = await hasBin('infisical');
  rows.push({ id: '(meta)', vars: infisicalBin ? 'infisical-cli-ok' : 'infisical-cli-MISSING', warnings: infisicalAuth === 'ok' ? '' : 'INFISICAL_* auth env vars missing' });
  for (const [id, entry] of Object.entries(reg.subprojects)) {
    try {
      const { env, warnings } = await resolveEnv({ subprojectId: id, cwd: metaRoot });
      const pmOk = await hasBin(entry.packageManager);
      rows.push({ id, vars: String(Object.keys(env).length), warnings: warnings.length ? warnings.join('; ').slice(0, 80) : (pmOk ? 'ok' : `missing-pm:${entry.packageManager}`) });
    } catch (e) {
      rows.push({ id, vars: 'err', warnings: String((e as Error).message).slice(0, 80) });
    }
  }
  if (json) printJson(rows); else printTable(rows);
  const hasAuthFailure = rows.some(r => r.warnings.includes('auth'));
  return hasAuthFailure ? 3 : 0;
}

async function hasBin(name: string): Promise<boolean> {
  try { await execa('which', [name]); return true; } catch { return false; }
}
```

```typescript
export async function runPassthroughCommand(id: string, args: string[]): Promise<number> {
  const metaRoot = findMetaRoot();
  const reg = loadRegistry(path.join(metaRoot, 'minion.json'));
  const entry = getSubproject(reg, id);
  const { env } = await resolveEnv({ subprojectId: id, cwd: metaRoot });
  const child = runCommand(args.join(' '), { cwd: path.join(metaRoot, entry.path), env });
  const r = await child;
  return r.exitCode ?? 1;
```

```typescript
#!/usr/bin/env node
import { Command } from 'commander';
import { devCommand } from './commands/dev';
import { buildCommand } from './commands/build';
import { testCommand } from './commands/test';
import { checkCommand } from './commands/check';
import { runPassthroughCommand } from './commands/run';
import { fanoutCommand } from './commands/fanout';
import { statusCommand } from './commands/status';
import { doctorCommand } from './commands/doctor';
import { syncEnvCommand } from './commands/sync-env';
import { rotateEnvCommand } from './commands/rotate-env';
import { infisicalCommand } from './commands/infisical';
import { linkCommand } from './commands/link';
import { listCommand } from './commands/list';
import { branchCommand } from './commands/branch';

export async function main(argv: string[] = process.argv): Promise<number> {
  const program = new Command();
  program.name('minion').description('Minion meta-repo CLI').version('0.1.0');

  program.command('dev').argument('<id>').option('--all').action(async (id, opts) => {
    process.exit(opts.all ? await fanoutCommand('dev') : await devCommand(id));
  });
  program.command('build').argument('<id>').option('--all').action(async (id, opts) => {
    process.exit(opts.all ? await fanoutCommand('build') : await buildCommand(id));
  });
  program.command('test').argument('<id>').option('--all').action(async (id, opts) => {
    process.exit(opts.all ? await fanoutCommand('test') : await testCommand(id));
  });
  program.command('check').argument('<id>').option('--all').action(async (id, opts) => {
    process.exit(opts.all ? await fanoutCommand('check') : await checkCommand(id));
  });
  program.command('run').argument('<id>').argument('<cmd...>').action(async (id, cmd) => {
    process.exit(await runPassthroughCommand(id, cmd));
  });
  program.command('status').option('--json').action(async (opts) => process.exit(await statusCommand(!!opts.json)));
  program.command('doctor').option('--json').action(async (opts) => process.exit(await doctorCommand(!!opts.json)));
  program.command('sync-env').argument('<id>').action(async (id) => process.exit(await syncEnvCommand(id)));
  program.command('rotate-env').argument('<id>').action(async (id) => process.exit(await rotateEnvCommand(id)));
  program.command('infisical').argument('<id>').action(async (id) => process.exit(await infisicalCommand(id)));
  program.command('link').argument('<id>').action(async (id) => process.exit(await linkCommand(id, false)));
  program.command('unlink').argument('<id>').action(async (id) => process.exit(await linkCommand(id, true)));
  program.command('list').option('--json').action(async (opts) => process.exit(await listCommand(!!opts.json)));
  program.command('branch').argument('<id>').action(async (id) => process.exit(await branchCommand(id)));

  // Shorthand: minion <id> <cmd...>  →  run
  // Commander doesn't support this out of the box; we hook in manually
  const knownCommands = new Set(program.commands.map(c => c.name()));
  const rawArgs = argv.slice(2);
  if (rawArgs[0] && !knownCommands.has(rawArgs[0]) && !rawArgs[0].startsWith('-')) {
    const id = rawArgs[0];
    const cmd = rawArgs.slice(1);
    if (cmd.length > 0) {
      process.exit(await runPassthroughCommand(id, cmd));
    }
  }

  await program.parseAsync(argv);
  return 0;
}

main().catch(err => {
  console.error('minion:', err.message ?? err);
  process.exit(1);
});
```
